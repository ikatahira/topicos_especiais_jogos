💻 Código JavaScript — Adaptive Learning Engine
📄 adaptive-learning.js
// ═══════════════════════════════════════════════
// ADAPTIVE LEARNING ENGINE — versão simplificada
// Regra: +2 acertos seguidos → sobe nível
//        +2 erros  seguidos → desce nível
// ═══════════════════════════════════════════════

const LEVELS = [
  { id: 'easy',   label: 'Fácil',  pts: 10 },
  { id: 'medium', label: 'Médio',  pts: 20 },
  { id: 'hard',   label: 'Difícil', pts: 30 },
];

class AdaptiveEngine {
  constructor(questions) {
    this.questions  = questions;   // 15 perguntas
    this.levelIndex = 0;          // começa no fácil
    this.streak     = 0;          // acertos seguidos
    this.failStreak = 0;          // erros seguidos
    this.history    = [];          // trajetória
    this.score      = 0;
    this.qIndex     = 0;
  }

  currentLevel() {
    return LEVELS[this.levelIndex];
  }

  nextQuestion() {
    // Busca pergunta do nível atual não respondida
    const lvl = this.currentLevel().id;
    const pool = this.questions
      .filter(q => q.level === lvl && !q.used);
    if (!pool.length) return null; // esgotou
    return pool[0];
  }

  answer(isCorrect) {
    const q = this.nextQuestion();
    if (q) q.used = true;

    if (isCorrect) {
      this.score      += this.currentLevel().pts;
      this.streak++;
      this.failStreak = 0;
      // Sobe nível ao acertar 2 seguidas
      if (this.streak >= 2 && this.levelIndex < 2) {
        this.levelIndex++;
        this.streak = 0;
      }
    } else {
      this.failStreak++;
      this.streak = 0;
      // Desce nível ao errar 2 seguidas
      if (this.failStreak >= 2 && this.levelIndex > 0) {
        this.levelIndex--;
        this.failStreak = 0;
      }
    }

    // Registra na trajetória
    this.history.push({
      q: this.qIndex++,
      level: this.levelIndex,
      correct: isCorrect,
    });
  }
}

⚖️ ADAPTIVE vs LINEAR — Diferenças Fundamentais
📋 Linear (tradicional)
• Todos os alunos veem as mesmas perguntas
• Ordem fixa: Q1→Q2→...→Q15
• Sem ajuste de dificuldade
• Rápido demais para alguns, lento para outros
• Não identifica lacunas individuais
🔄 Adaptive (nosso)
• Trajetória única por aluno
• Dificuldade ajusta em tempo real
• Mantém aluno na Zona Proximal
• Detecta e reforça pontos fracos
• Maximiza aprendizado por tempo gasto
